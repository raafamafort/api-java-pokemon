CREATE TABLE pokemon (
    id NUMBER(19) GENERATED BY DEFAULT ON NULL AS IDENTITY,
    nome VARCHAR2(255) NOT NULL,
    altura VARCHAR2(255) NOT NULL,
    peso VARCHAR2(255),
    categoria VARCHAR2(255) NOT NULL,
    habilidades VARCHAR2(255) NOT NULL,
    ativo NUMBER(1) NOT NULL,
    CONSTRAINT pk_pokemon PRIMARY KEY (id)
);
